% We first define the type of document this is
% article is the most useful type - there are other 
% including book and beamer for presentations
\documentclass{article}

\usepackage{booktabs}
\usepackage{graphicx}

\author{John Doe}
\title{AirBnB and Random Forests}

\begin{document}

\maketitle

\section{Introduction}

% input another file with the \input command

\input{intro}

\section{Results}

\input{plot}

\bibliographystyle{plain}
\bibliography{references}

\end{document}
